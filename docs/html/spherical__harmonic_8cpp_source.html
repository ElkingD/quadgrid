105 
106  Ylm[1] = table.Y10*rUnit[2];

112  size_t l_index_m1 = 1;
113  size_t l_index_m2 = 0;
114 
115  for (size_t l = 2; l <= lmax; l++)
116  {

122

124  double fact2 = table.C4[l-1]*table.B2[l-1+m]*table.B2[l-1-m];
125  Ylm[index1] = fact1*Ylm[index2] - fact2*Ylm[index3];
126  }
127

129  index1 = l_index + l-1;
